    <script>
         //Utilizando bucles, el símbolo &nbsp y la etiqueta <br>
        /*
            i=0   0    *     1(*)  6(blanco)
            i=1       ***    3(*)  4(blanco)
            i=2      *****   5(*)  2(blanco)
            i=3     *******  7(*)  0(blanco)
            i=4      *****   5(*)  2(blanco)
            i=5       ***    3(*)  4(blanco)
            i=6        *     1(*)  6(blanco)

                    filas = n*2 + 1
        */
    function dibujarRombo() {
        let numeroLineas= Number(prompt("Ingrese el número de líneas del rombo: ")); // Se pide al usuario que introduzca el número de filas del rombo y lo guardo en 'numeroLineas'
        let output = '';

        // Primera mitad del rombo
        for (let ancho = 0; ancho < numeroLineas; ancho++) {  //en el bucle for indico el ancho, es decir, la distancia que recorre de izquierda a derecha
            for (let alto = 0; alto < numeroLineas - ancho - 1; alto++) { //el alto es la distancia que recorre de arriba a abajo y se ponen los espacio es blanco con '&nbsp'
                output += '&nbsp;&nbsp;';
            }
            for (let pintar = 0; pintar < 2 * ancho + 1; pintar++) { //pintar es la cantidad de asteriscos que se van a poner en cada fila
                output += "*";
            }
            output += '<br>';
        }

        // Segunda mitad del rombo
        //La parte inferior sigue el mismo procedimiento que la parte superior, pero en orden inverso. 
        //En vez de empezar con 0 asteriscos y terminar con 5, se empieza con 5 y se termina con 0. Se van restando por fila
        for (let ancho = numeroLineas - 2; ancho >= 0; ancho--) {
            for (let alto = 0; alto < numeroLineas - ancho - 1; alto++) {
                output += '&nbsp;&nbsp;';
            }
            for (let pintar = 0; pintar < 2 * ancho + 1; pintar++) {
                output += "*";
            }
            output += '<br>';
        }
        document.body.innerHTML = output;
    }

    dibujarRombo(); //se llama a la función para que se ejecute
    </script>
